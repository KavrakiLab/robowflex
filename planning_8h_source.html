99  /** \brief Initialize the planner pool.
100  * Forwards template arguments \a Args to the initializer of the templated planner \a P. Assumes that
101  * the constructor of the planner takes \a robot_ and \a name_.
102  * \param[in] args Arguments to initializer of planner \a P.
103  * \tparam P The robowflex::Planner to pool.
104  * \tparam Args Argument types to initializer of planner \a P.
105  * \return True on success, false on failure.
106  */
107  template <typename P, typename... Args>
108  bool initialize(Args &&... args)
109  {
110  for (unsigned int i = 0; i < pool_.getThreadCount(); ++i)
111  {
112  auto planner = std::make_shared<P>(robot_, name_);
113 
114  if (!planner->initialize(std::forward<Args>(args)...))
115  return false;
116 
117  planners_.emplace(std::move(planner));
118  }
119 
120  return true;
121  }
