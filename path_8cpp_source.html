72 double robowflex::path::getSmoothness(const robot_trajectory::RobotTrajectory &path, const PathMetric &metric)
73 {
74  double smoothness = 0.0;
75 
76  auto distance =
77  (metric) ? metric : [](const robot_state::RobotState &a, const robot_state::RobotState &b) {
78  return a.distance(b);
79  };
80 
81  // compute smoothness
82  if (path.getWayPointCount() > 2)
83  {
84  double a = distance(path.getWayPoint(0), path.getWayPoint(1));
85  for (std::size_t k = 2; k < path.getWayPointCount(); ++k)
86  {
87  // view the path as a sequence of segments, and look at the triangles it forms:
88  // s1
89  // /\ s4
90  // a / \ b |
91  // / \ |
92  // /......\_______|
93  // s0 c s2 s3
94  //
95 
96  // use Pythagoras generalized theorem to find the cos of the angle between segments a and b
97  double b = distance(path.getWayPoint(k - 1), path.getWayPoint(k));
98  double cdist = distance(path.getWayPoint(k - 2), path.getWayPoint(k));
99  double acosValue = (a * a + b * b - cdist * cdist) / (2.0 * a * b);
100  if (acosValue > -1.0 && acosValue < 1.0)
101  {
102  // the smoothness is actually the outside angle of the one we compute
103  double angle = (boost::math::constants::pi<double>() - acos(acosValue));
104 
105  // and we normalize by the length of the segments
106  double u = 2.0 * angle; /// (a + b);
107  smoothness += u * u;
108  }
109 
110  a = b;
111  }
112 
113  smoothness /= path.getWayPointCount();
114  }
115 
116  return smoothness;
117 }
