163 void MoveGroupHelper::moveGroupGoalCallback(const moveit_msgs::MoveGroupActionGoal &msg)

177 void MoveGroupHelper::moveGroupResultCallback(const moveit_msgs::MoveGroupActionResult &msg)
178 {
179  const std::string &id = msg.status.goal_id.id;
180 
181  auto request = requests_.find(id);
182  if (request == requests_.end())
183  {
184  ROS_WARN("Intercepted unknown request response ID: `%s`", id.c_str());
185  return;
186  }
187 
188  ROS_DEBUG("Intercepted request response ID: `%s`", id.c_str());
189 
190  Action &action = request->second;
191  action.id = id;
192  action.success = msg.result.error_code.val == moveit_msgs::MoveItErrorCodes::SUCCESS;
193  action.time = msg.result.planning_time;
194  action.trajectory = msg.result.planned_trajectory;
195 
196  if (callback_)
197  callback_(action);
198 
199  requests_.erase(request);
200 }
