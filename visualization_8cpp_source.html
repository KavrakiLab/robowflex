146 void IO::RVIZHelper::addGeometryMarker(const std::string &name, const GeometryConstPtr &geometry,
147  const std::string &base_frame, const Eigen::Affine3d &pose,
148  const Eigen::Vector4d &color)
149 {
150  visualization_msgs::Marker marker;
151 
152  auto scale = geometry->getDimensions();
153  switch (geometry->getType())
154  {
155  case Geometry::ShapeType::BOX:
156  marker.type = visualization_msgs::Marker::CUBE;
157  break;
158  case Geometry::ShapeType::SPHERE:
159  marker.type = visualization_msgs::Marker::SPHERE;
160  scale[1] = scale[2] = scale[0]; // Copy radius to other dimensions
161  break;
162  case Geometry::ShapeType::CYLINDER:
163  marker.type = visualization_msgs::Marker::CYLINDER;
164  {
165  auto scale2 = scale;
166  scale[0] = scale[1] = scale2[0]; // Copy radius to first two (x & y)
167  scale[2] = scale2[1];
168  }
169  break;
170  case Geometry::ShapeType::MESH:
171  marker.type = visualization_msgs::Marker::MESH_RESOURCE;
172  marker.mesh_resource = geometry->getResource();
173  marker.mesh_use_embedded_materials = true;
174  break;
175  default:
176  ROS_ERROR("Unsupported geometry for marker.");
177  return;
178  };
179 
180  fillMarker(marker, base_frame, pose, color, scale);
181 
182  markers_.emplace(name, marker);
183 }

185 void IO::RVIZHelper::addGoalMarker(const std::string &name, const MotionRequestBuilder &request)
186 {
187  const auto &goals = request.getRequestConst().goal_constraints;
188  const auto &base_frame = "map";
189 
190  for (const auto &goal : goals)
191  {
192  auto color = getRandomColor();
193  color[3] = 0.7; // Make slightly transparent
194 
195  for (const auto &pg : goal.position_constraints)
196  {
197  const auto &pname = pg.link_name;
198 
199  Eigen::Affine3d pose = robot_->getLinkTF(pg.header.frame_id);
200  pose.translate(TF::vectorMsgToEigen(pg.target_point_offset));
201 
202  for (const auto &primitive :
203  boost::combine(pg.constraint_region.primitives, pg.constraint_region.primitive_poses))
204  {
205  shape_msgs::SolidPrimitive solid;
206  geometry_msgs::Pose solid_pose;
207  boost::tie(solid, solid_pose) = primitive;
208 
209  auto frame = pose * TF::poseMsgToEigen(solid_pose);
210  addGeometryMarker(name, Geometry::makeSolidPrimitive(solid), base_frame, frame, color);
211  addTextMarker(name, name + " " + pname, base_frame, frame * Eigen::Translation3d(0, 0, 0.1),
212  0.05);
213 
214  for (const auto &og : goal.orientation_constraints)
215  {
216  const auto &oname = og.link_name;
217  if (oname != pname)
218  continue;
219 
220  auto q = TF::quaternionMsgToEigen(og.orientation);
221 
222  Eigen::Vector3d scale = {0.1, 0.008, 0.003};
223  addArrowMarker(name, base_frame, frame * q, color, 1.5 * scale);
224 
225  const auto tolerances = {og.absolute_x_axis_tolerance, //
226  og.absolute_y_axis_tolerance, //
227  og.absolute_z_axis_tolerance};
228  const auto axes = {Eigen::Vector3d::UnitX(), //
229  Eigen::Vector3d::UnitY(), //
230  Eigen::Vector3d::UnitZ()};
231 
232  for (const auto &angles : boost::combine(tolerances, axes))
233  {
234  double value;
235  Eigen::Vector3d axis;
236  boost::tie(value, axis) = angles;
237 
238  auto q1 = TF::offsetOrientation(q, axis, value);
239  addArrowMarker(name, base_frame, frame * q1, color, scale);
240 
241  auto q2 = TF::offsetOrientation(q, axis, -value);
242  addArrowMarker(name, base_frame, frame * q2, color, scale);
243  }
244  }
