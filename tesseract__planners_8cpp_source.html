188 planning_interface::MotionPlanResponse
189 OMPLChainPlanner::plan(const SceneConstPtr &scene, const planning_interface::MotionPlanRequest &request)
190 {
191  planning_interface::MotionPlanResponse res;
192 
193  tesseract::tesseract_ros::KDLEnvPtr env = constructTesseractEnv(scene, getRobot());
194 
195  // Make the OMPL Context.
196  chain_interface_ =
197  std::make_shared<tesseract::tesseract_planning::ChainOmplInterface>(env, request.group_name);
198 
199  // Set the Motion Validator in chain_interface_.
200  if (settings_.use_continuous_validator)
201  {
202  auto mv = std::make_shared<tesseract::tesseract_planning::ContinuousMotionValidator>(
203  chain_interface_->spaceInformation(), env, request.group_name);
204  chain_interface_->setMotionValidator(mv);
205  }
206  chain_interface_->spaceInformation()->setup();
207  std::map<std::string, std::string> config;
208  config["longest_valid_segment_fraction"] = std::to_string(settings_.longest_valid_segment_fraction);
209  chain_interface_->spaceInformation()->params().setParams(config, true);
210  chain_interface_->spaceInformation()->setup();
211 
212  ompl::base::PlannerPtr planner;
213  auto it = known_planners_.find(request.planner_id);
214  if (it != known_planners_.end())
215  {
216  std::map<std::string, std::string> config; // TODO: get the actual version of this.
217  planner = it->second(chain_interface_->spaceInformation(), "tesseract_planner?", config);
218  }
219  else
220  {
221  ROS_ERROR("Unknown planner: '%s'", request.planner_id.c_str());
222  res.error_code_.val = moveit_msgs::MoveItErrorCodes::FAILURE;
223  return res;
224  }
225 
226  // Get start and goal pos.
227  // TODO: somehow handle pose targets, not just config targets.
228  robot_state::RobotState state = *robot_->getScratchState(); // make a copy of the states
229  robot_state::robotStateMsgToRobotState(request.start_state, state);
230 
231  std::vector<double> start_joints;
232  state.copyJointGroupPositions(request.group_name, start_joints);
233 
234  // TODO: extend Tesseract Planning to handle goal sampling?
235  // : already extened to handle arbitrary OMPL goals.
236  std::vector<double> goal_joints;
237  for (std::size_t i = 0; i < request.goal_constraints.size(); i++)
238  {
239  // We will only use the first constraint that has joints.
240  if (request.goal_constraints[i].joint_constraints.empty())
241  continue;
242 
243  std::map<std::string, double> positions;
244  for (const auto &joint_cnt : request.goal_constraints[i].joint_constraints)
245  positions[joint_cnt.joint_name] = joint_cnt.position;
246 
247  state.setVariablePositions(positions);
248  state.update();
249 
250  state.copyJointGroupPositions(request.group_name, goal_joints);
251  }
252 
253  if (goal_joints.empty())
254  {
255  ROS_ERROR("We can only handle Joint Goals at the moment. Sorry.");
256  res.error_code_.val = moveit_msgs::MoveItErrorCodes::FAILURE;
257  return res;
258  }
259 
260  tesseract::tesseract_planning::OmplPlanParameters params;
261  params.planning_time = request.allowed_planning_time;
262  params.simplify = settings_.simplify_solutions;
263 
264  // Call plan.
265  ompl::time::point start = ompl::time::now();
266  auto maybe_path = chain_interface_->plan(planner, start_joints, goal_joints, params);
267  ompl::time::duration total_time = ompl::time::now() - start;
268 
269  if (maybe_path)
270  {
271  const ompl::geometric::PathGeometric &path = *maybe_path;
272  res.trajectory_.reset(new robot_trajectory::RobotTrajectory(robot_->getModel(), request.group_name));
273  for (std::size_t i = 0; i < path.getStateCount(); i++)
274  {
275  state.setJointGroupPositions(
276  request.group_name,
277  path.getState(i)->as<ompl::base::RealVectorStateSpace::StateType>()->values);
278  state.update();
279  res.trajectory_->addSuffixWayPoint(state, 0.0);
280  }
281 
282  res.planning_time_ = ompl::time::seconds(total_time);
283  res.error_code_.val = moveit_msgs::MoveItErrorCodes::SUCCESS;
284  return res;
285  }
286  else
287  {
288  ROS_WARN("Planning failed");
289  res.error_code_.val = moveit_msgs::MoveItErrorCodes::PLANNING_FAILED;
290  return res;
291  }
292 }
