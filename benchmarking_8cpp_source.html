74 void Benchmarker::Results::computeMetric(planning_interface::MotionPlanResponse &run, Run &metrics)
75 {
76  const robot_trajectory::RobotTrajectory &p = *run.trajectory_;
77  const planning_scene::PlanningScene &s = *scene->getSceneConst();
78 
79  if (options.options & MetricOptions::WAYPOINTS)
80  metrics.metrics["waypoints"] = (int)p.getWayPointCount();
81 
82  if (options.options & MetricOptions::PATH)
83  p.getRobotTrajectoryMsg(metrics.path);
84 
85  if (options.options & MetricOptions::LENGTH)
86  {
87  double length = 0.0;
88  // compute path length
89  for (std::size_t k = 1; k < p.getWayPointCount(); ++k)
90  length += p.getWayPoint(k - 1).distance(p.getWayPoint(k));
91 
92  metrics.metrics["length"] = length;
93  }
94 
95  // compute correctness and clearance
96  if (options.options & (MetricOptions::CORRECT | MetricOptions::CLEARANCE))
97  {
98  bool correct = true;
99  double clearance = 0.0;
100 
101  collision_detection::CollisionRequest request;
102  for (std::size_t k = 0; k < p.getWayPointCount(); ++k)
103  {
104  if (options.options & MetricOptions::CORRECT)
105  {
106  collision_detection::CollisionResult result;
107  s.checkCollisionUnpadded(request, result, p.getWayPoint(k));
108 
109  if (result.collision)
110  correct = false;
111 
112  if (!p.getWayPoint(k).satisfiesBounds())
113  correct = false;
114  }
115 
116  if (options.options & MetricOptions::CLEARANCE)
117  {
118  double d = s.distanceToCollisionUnpadded(p.getWayPoint(k));
119  if (d > 0.0) // in case of collision, distance is negative
120  clearance += d;
121  }
122  }
123 
124  if (options.options & MetricOptions::CORRECT)
125  metrics.metrics["correct"] = correct;
126 
127  if (options.options & MetricOptions::CLEARANCE)
128  {
129  clearance /= (double)p.getWayPointCount();
130  metrics.metrics["clearance"] = clearance;
131  }
132  }
133 
134  if (options.options & MetricOptions::SMOOTHNESS)
135  {
136  double smoothness = 0.0;
137 
138  // compute smoothness
139  if (p.getWayPointCount() > 2)
140  {
141  double a = p.getWayPoint(0).distance(p.getWayPoint(1));
142  for (std::size_t k = 2; k < p.getWayPointCount(); ++k)
143  {
144  // view the path as a sequence of segments, and look at the triangles it forms:
145  // s1
146  // /\ s4
147  // a / \ b |
148  // / \ |
149  // /......\_______|
150  // s0 c s2 s3
151  //
152 
153  // use Pythagoras generalized theorem to find the cos of the angle between segments a and b
154  double b = p.getWayPoint(k - 1).distance(p.getWayPoint(k));
155  double cdist = p.getWayPoint(k - 2).distance(p.getWayPoint(k));
156  double acosValue = (a * a + b * b - cdist * cdist) / (2.0 * a * b);
157  if (acosValue > -1.0 && acosValue < 1.0)
158  {
159  // the smoothness is actually the outside angle of the one we compute
160  double angle = (boost::math::constants::pi<double>() - acos(acosValue));
161 
162  // and we normalize by the length of the segments
163  double u = 2.0 * angle; /// (a + b);
164  smoothness += u * u;
165  }
166 
167  a = b;
168  }
169 
170  smoothness /= (double)p.getWayPointCount();
171  }
172 
173  metrics.metrics["smoothness"] = smoothness;
174  }
175 }

189 void Benchmarker::benchmark(const std::vector<BenchmarkOutputterPtr> &outputs, const Options &options)
190 {
191  unsigned int count = 0;
192  const unsigned int total = requests_.size() * options.runs;
193 
194  for (const auto &request : requests_)
195  {
196  const auto &name = request.first;
197  auto &scene = std::get<0>(request.second);
198  auto &planner = std::get<1>(request.second);
199  const auto &builder = std::get<2>(request.second);
200  std::vector<moveit_msgs::RobotTrajectory> trajectories;
201 
202  Results results(name, scene, planner, builder, options);
203 
204  for (unsigned int j = 0; j < options.runs; ++j)
205  {
206  ros::WallTime start;
207 
208  start = ros::WallTime::now();
209  planning_interface::MotionPlanResponse response = planner->plan(scene, builder->getRequest());
210  double time = (ros::WallTime::now() - start).toSec();
211 
212  results.addRun(j, time, response);
213  ROS_INFO("BENCHMARKING: [ %u / %u ] Completed", ++count, total);
214  }
215 
216  results.finish = IO::getDate();
217 
218  for (BenchmarkOutputterPtr output : outputs)
219  output->dumpResult(results);
220  }
221 }
