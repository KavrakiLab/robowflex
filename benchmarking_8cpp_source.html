75 void Benchmarker::Results::computeMetric(planning_interface::MotionPlanResponse &run, Run &metrics)
76 {
77  const robot_trajectory::RobotTrajectory &p = *run.trajectory_;
78 
79  if (options.options & MetricOptions::WAYPOINTS)
80  metrics.metrics["waypoints"] = (int)p.getWayPointCount();
81 
82  if (options.options & MetricOptions::PATH)
83  p.getRobotTrajectoryMsg(metrics.path);
84 
85  if (options.options & MetricOptions::LENGTH)
86  metrics.metrics["length"] = path::getLength(p);
87 
88  if (options.options & MetricOptions::CORRECT)
89  metrics.metrics["correct"] = path::isCorrect(p, scene);
90 
91  if (options.options & MetricOptions::CLEARANCE)

93

110 void Benchmarker::benchmark(const std::vector<BenchmarkOutputterPtr> &outputs, const Options &options)
111 {
112  unsigned int count = 0;
113  const unsigned int total = requests_.size() * options.runs;
114 
115  for (const auto &request : requests_)
116  {
117  const auto &name = request.first;
118  auto &scene = std::get<0>(request.second);
119  auto &planner = std::get<1>(request.second);
120  const auto &builder = std::get<2>(request.second);
121  std::vector<moveit_msgs::RobotTrajectory> trajectories;
122 
123  Results results(name, scene, planner, builder, options);
124 
125  for (unsigned int j = 0; j < options.runs; ++j)
126  {
127  ros::WallTime start;
128 
129  start = ros::WallTime::now();
130  planning_interface::MotionPlanResponse response = planner->plan(scene, builder->getRequest());
131  double time = (ros::WallTime::now() - start).toSec();
132 
133  results.addRun(j, time, response);
134  ROS_INFO("BENCHMARKING: [ %u / %u ] Completed", ++count, total);
135  }
136 
137  results.finish = IO::getDate();
138 
139  for (const BenchmarkOutputterPtr& output : outputs)
140  output->dumpResult(results);
141  }
142 }

158 void JSONBenchmarkOutputter::dumpResult(const Benchmarker::Results &results)
159 {
160  if (not is_init_)
161  {
162  IO::createFile(outfile_, file_);
163  outfile_ << "{";
164  // TODO: output specific information about the scene and planner structs?
165 
166  is_init_ = true;
167  }
168  else
169  outfile_ << ",";
170 
171  outfile_ << "\"" << results.name << "\":[";
172 
173  for (size_t i = 0; i < results.runs.size(); i++)
174  {
175  const Benchmarker::Results::Run &run = results.runs[i];
176  outfile_ << "{";
177 
178  outfile_ << R"("name": "run_)" << run.num << "\",";
179  outfile_ << "\"time\":" << run.time << ",";
180  outfile_ << "\"success\":" << run.success;
181 
182  for (const auto &metric : run.metrics)
183  outfile_ << ",\"" << metric.first
184  << "\":" << boost::apply_visitor(Benchmarker::Results::Run::toString(), metric.second);
185 
186  outfile_ << "}";
187 
188  // Write the command between each run.
189  if (i != results.runs.size() - 1)
190  outfile_ << "," << std::endl;
191  }
192 
193  outfile_ << "]";
194 }
