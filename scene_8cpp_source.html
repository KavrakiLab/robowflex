14  /** \brief The actual plugin loader for collision plugins.
15  * Heavily inspired by code in moveit_ros/planning. */
16  class Scene::CollisionPluginLoader
17  {
18  /** \brief The pluginlib loader for collision detection plugins.
19  */
20  using PluginLoader = pluginlib::ClassLoader<collision_detection::CollisionPlugin>;
21 
22  public:
23  /** \brief Constructor. Attempts to create the pluginlib loader for collision plugins.
24  */
25  CollisionPluginLoader()
26  {
27  try
28  {
29  loader_.reset(new PluginLoader("moveit_core", "collision_detection::CollisionPlugin"));
30  }
31  catch (pluginlib::PluginlibException &e)
32  {
33  ROS_ERROR("Unable to construct collision plugin loader. Error: %s", e.what());
34  }
35  }
36 
37  /** \brief Attempts to load the collision detector plugin by the given \a name.
38  * Saves the plugin in an internal map and returns it if found.
39  * \param[in] name Name of the plugin to load.
40  * \return An allocated collision plugin.
41  */
42  collision_detection::CollisionPluginPtr load(const std::string &name)
43  {
44  collision_detection::CollisionPluginPtr plugin;
45 
46  try
47  {
48  plugin.reset(loader_->createUnmanagedInstance(name));
49  plugins_[name] = plugin;
50  }
51  catch (pluginlib::PluginlibException &ex)
52  {
53  ROS_ERROR_STREAM("Exception while loading " << name << ": " << ex.what());
54  }
55 
56  return plugin;
57  }
58 
59  /** \brief Loads a collision detector into a planning scene instance.
60  * \param[in] name the plugin name
61  * \param[in] scene the planning scene instance.
62  * \param[in] exclusive If true, the new collision detector is the only one.
63  * \return True if the new collision detector is added to the scene.
64  */
65  bool activate(const std::string &name, const planning_scene::PlanningScenePtr &scene, bool exclusive)
66  {
67  auto it = plugins_.find(name);
68  if (it == plugins_.end())
69  {
70  collision_detection::CollisionPluginPtr plugin = load(name);
71  if (plugin)
72  return plugin->initialize(scene, exclusive);
73  }
74  else if (it->second)
75  return it->second->initialize(scene, exclusive);
76 
77  return false;
78  }
79 
80  private:
81  std::shared_ptr<PluginLoader> loader_; // The pluginlib loader.
82  std::map<std::string, collision_detection::CollisionPluginPtr> plugins_; ///< Loaded plugins.
83  };
84 } // namespace robowflex
