33 RobotPose TF::createPoseXYZ(const Eigen::Ref<const Eigen::Vector3d> &translation,

35 {
36  RobotPose pose = RobotPose::Identity();

38

40  Eigen::AngleAxisd(rotation[1], Eigen::Vector3d::UnitY()) *

43
