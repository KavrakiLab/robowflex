285  double remap(double a1, double a2, double av, double b1, double b2)
286  {
287  double dat = fabs(a2 - a1);
288  double dav = fabs(av - a1);
289 
290  double dbt = fabs(b2 - b1);
291  double dbv = dbt * (dav / dat);
292 
293  double bv = b1 + dbv;
294 
295  return bv;
296  }
297  void colormap(double s, double &r, double &g, double &b, unsigned int size, const Entry map[])
298  {
299  for (unsigned int i = 1; i < size; ++i)
300  {
301  const Entry &p = map[i - 1];
302  const Entry &n = map[i];
303 
304  if (s < n.s)
305  {
306  r = remap(p.s, n.s, s, p.r, n.r);
307  g = remap(p.s, n.s, s, p.g, n.g);
308  b = remap(p.s, n.s, s, p.b, n.b);
309  return;
310  }
311  }
