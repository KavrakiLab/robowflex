87 const std::string IO::resolvePackage(const std::string &path)
88 {
89  if (path.empty())
90  return "";
91 
92  const std::string prefix = "package://";
93 
94  boost::filesystem::path file;
95  if (isPrefix(prefix, path))
96  {
97  boost::filesystem::path subpath(path.substr(prefix.length(), path.length() - 1));
98  const std::string package_name = (*subpath.begin()).string();
99 
100  const std::string package = ros::package::getPath(package_name);
101  if (package.empty())
102  {
103  ROS_WARN("Package `%s` does not exist.", package_name.c_str());
104  return "";
105  }
106 
107  file = package;
108  for (auto it = ++subpath.begin(); it != subpath.end(); ++it)
109  file /= *it;
110  }
111  else
112  file = path;
113 
114  return expandPath(file).string();
115 }

338  {
