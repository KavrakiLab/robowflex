40 bool MotionRequestBuilder::setConfig(const std::string &requested_config)
41 {
42  const auto &configs = planner_->getPlannerConfigs();
43 
44  std::vector<std::reference_wrapper<const std::string>> matches;
45  for (const auto &config : configs)
46  {
47  if (config.find(requested_config) != std::string::npos)
48  matches.emplace_back(config);
49  }
50 
51  if (matches.empty())
52  return false;
53 
54  const auto &found =
55  std::min_element(matches.begin(), matches.end(),
56  [](const std::string &a, const std::string &b) { return a.size() < b.size(); });
57 
58  request_.planner_id = *found;
59  ROS_INFO("Using planner: %s", request_.planner_id.c_str());
60  return true;
61 }
