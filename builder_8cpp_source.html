44 bool MotionRequestBuilder::setConfig(const std::string &requested_config)
45 {
46  const auto &configs = planner_->getPlannerConfigs();
47 
48  std::vector<std::reference_wrapper<const std::string>> matches;
49  for (const auto &config : configs)
50  {
51  if (config.find(requested_config) != std::string::npos)
52  matches.emplace_back(config);
53  }
54 
55  if (matches.empty())
56  return false;
57 
58  const auto &found =
59  std::min_element(matches.begin(), matches.end(),
60  [](const std::string &a, const std::string &b) { return a.size() < b.size(); });
61 
62  request_.planner_id = *found;
63  ROS_INFO("Using planner: %s", request_.planner_id.c_str());
64  return true;
65 }

122 void MotionRequestBuilder::addGoalRotaryTile(const std::string &ee_name, const std::string &base_name,
123  const RobotPose &pose, const GeometryConstPtr &geometry,
124  const Eigen::Quaterniond &orientation,
125  const Eigen::Vector3d &tolerances, const RobotPose &offset,
126  const Eigen::Vector3d &axis, unsigned int n)
127 {
128  double pi2 = 2 * boost::math::constants::pi<double>();
129 
130  for (double angle = 0; angle < pi2; angle += pi2 / n)
131  {
132  Eigen::Quaterniond rotation(Eigen::AngleAxisd(angle, axis));
133  RobotPose newPose = pose * rotation * offset;
134  Eigen::Quaterniond newOrientation(rotation * orientation);
135 
136  setGoalRegion(ee_name, base_name, newPose, geometry, newOrientation, tolerances);
137  }
