158 bool Robot::loadKinematics(const std::string &name)
159 {
160  if (imap_.find(name) != imap_.end())
161  return true;
162 
163  robot_model::SolverAllocatorFn allocator = kinematics_->getLoaderFunction(loader_->getSRDF());
164 
165  const auto &groups = kinematics_->getKnownGroups();
166  if (groups.empty())
167  {
168  ROS_WARN("No kinematics plugins defined. Fill and load kinematics.yaml!");
169  return false;
170  }
171 
172  if (!model_->hasJointModelGroup(name) || std::find(groups.begin(), groups.end(), name) == groups.end())
173  {
174  ROS_WARN("No JMG or Kinematics defined for `%s`!", name.c_str());
175  return false;
176  }
177 
178  robot_model::JointModelGroup *jmg = model_->getJointModelGroup(name);
179  kinematics::KinematicsBasePtr solver = allocator(jmg);
180 
181  if (solver)
182  {
183  std::string error_msg;
184  if (solver->supportsGroup(jmg, &error_msg))
185  imap_[name] = allocator;
186 
187  else
188  {
189  ROS_ERROR("Kinematics solver %s does not support joint group %s. Error: %s",
190  typeid(*solver).name(), name.c_str(), error_msg.c_str());
191  return false;
192  }
193  }
194  else
195  {
196  ROS_ERROR("Kinematics solver could not be instantiated for joint group `%s`.", name.c_str());
197  return false;
198  }
199 
200  auto timeout = kinematics_->getIKTimeout();
201  auto attempts = kinematics_->getIKAttempts();
202 
203  jmg->setDefaultIKTimeout(timeout[name]);
204  jmg->setDefaultIKAttempts(attempts[name]);
205 
206  model_->setKinematicsAllocators(imap_);
207 
208  return true;
209 }

351  YAML::Node addLinkGeometry(const urdf::GeometrySharedPtr &geometry)
352  {
353  YAML::Node node;
354  switch (geometry->type)
355  {
356  case urdf::Geometry::MESH:
357  {
358  const auto &mesh = static_cast<urdf::Mesh *>(geometry.get());
359  node["type"] = "mesh";
360  node["resource"] = IO::resolvePath(mesh->filename);
361 
362  if (mesh->scale.x != 1 || mesh->scale.y != 1 || mesh->scale.z != 1)
363  {
364  node["dimensions"] = std::vector<double>({mesh->scale.x, mesh->scale.y, mesh->scale.z});
365  ROBOWFLEX_YAML_FLOW(node["dimensions"]);
366  }
367  break;
368  }
369  case urdf::Geometry::BOX:
370  {
371  const auto &box = static_cast<urdf::Box *>(geometry.get());
372  node["type"] = "box";
373  node["dimensions"] = std::vector<double>({box->dim.x, box->dim.y, box->dim.z});
374  ROBOWFLEX_YAML_FLOW(node["dimensions"]);
375  break;
376  }
377  case urdf::Geometry::SPHERE:
378  {
379  const auto &sphere = static_cast<urdf::Sphere *>(geometry.get());
380  node["type"] = "sphere";
381  node["dimensions"] = std::vector<double>({sphere->radius});
382  ROBOWFLEX_YAML_FLOW(node["dimensions"]);
383  break;
384  }
385  case urdf::Geometry::CYLINDER:
386  {
387  const auto &cylinder = static_cast<urdf::Cylinder *>(geometry.get());
388  node["type"] = "cylinder";
389  node["dimensions"] = std::vector<double>({cylinder->length, cylinder->radius});
390  ROBOWFLEX_YAML_FLOW(node["dimensions"]);
391  break;
392  }
393  default:
394  break;
395  }
396 
397  return node;
398  }

511  add["name"] = link->name;
512 
513  if (!visual.IsNull())

519  link_geometry.push_back(add);
520  }
521  }

534 bool Robot::dumpPathTransforms(const robot_trajectory::RobotTrajectory &path, const std::string &filename,
535  double fps, double threshold) const
536 {
537  YAML::Node node, values;
538  const double rate = 1.0 / fps;
539 
540  // Find the total duration of the path.
541  const std::deque<double> &durations = path.getWayPointDurations();
542  double total_duration = std::accumulate(durations.begin(), durations.end(), 0.0);
543 
544  const auto &urdf = model_->getURDF();
545 
546  robot_state::RobotStatePtr previous, state(new robot_state::RobotState(model_));
547 
548  for (double duration = 0.0, delay = 0.0; duration <= total_duration; duration += rate, delay += rate)
549  {
550  YAML::Node point;
551 
552  path.getStateAtDurationFromStart(duration, state);
553  if (previous && state->distance(*previous) < threshold)
554  continue;
555 
556  state->update();
557  for (const auto &link_name : model_->getLinkModelNames())
558  {
559  const auto &urdf_link = urdf->getLink(link_name);
560  if (urdf_link->visual)
561  {
562  const auto &link = model_->getLinkModel(link_name);
563  RobotPose tf =
564  state->getGlobalLinkTransform(link) * urdfPoseToEigen(urdf_link->visual->origin);
565  point[link->getName()] = IO::toNode(TF::poseEigenToMsg(tf));
566  }
567  }
568 
569  YAML::Node value;
570  value["point"] = point;
571  value["duration"] = delay;
572  values.push_back(value);
573 
574  delay = 0;
575 
576  if (!previous)
577  previous.reset(new robot_state::RobotState(model_));
578 
579  *previous = *state;
580  }
581 
582  node["transforms"] = values;
583  node["fps"] = fps;
584 
585  return IO::YAMLToFile(node, filename);
586 }
